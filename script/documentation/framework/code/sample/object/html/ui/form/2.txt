$MyForm = new sPHP\HTML\UI\Form(
	null,
	null,
	[
		new InputValidation("Name", true, VALIDATION_TYPE_ALPHABETIC),
		new InputValidation("Email", true, VALIDATION_TYPE_EMAIL),
		new InputValidation("Comment", true),
	]
);

if($MyForm->Verify()){
	print "The form submission went successful.";
}
else{
	print "There was a problem with the form! The error is: " . $MyForm->ErrorMessage();
}
