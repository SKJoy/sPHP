$MyForm = new sPHP\HTML\UI\Form(
	null,
	null,
	[
		new InputValidation("Name", true, VALIDATION_TYPE_ALPHABETIC),
		new InputValidation("Email", true, VALIDATION_TYPE_EMAIL),
		new InputValidation("Comment", true),
	]
);

if($MyForm->Authenticate()){
	print "Source of the form is authentic.";

	if($MyForm->ValidateInput()){
		print "Form input(s) are okay.";
	}
	else{
		print "Validation error: " . $MyForm->ErrorMessage();
	}
}
else{
	print "Authentication error: " . $MyForm->ErrorMessage();
}
